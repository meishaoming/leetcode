```py
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:

        def dfs(i):
            if i == len(nums):
                ans.append(nums[:])
                return
            
            for t in range(i, len(nums)):
                nums[i], nums[t] = nums[t], nums[i]
                dfs(i+1)
                nums[i], nums[t] = nums[t], nums[i]

        ans = []
        dfs(0)
        return ans
```
